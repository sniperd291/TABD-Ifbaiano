CREATE DATABASE CRIME;
USE DATABASE CRIME;

CREATE TABLE CRIMINOSO(
ID CRIMINOSO VARCHAR(50),
NOME VARCHAR(30),
CPF VARCHAR(8),
RG INT,
);

CREATE TABLE VITIMA(
ID VITIMA INT,
NOME VARCHAR(30),
);

CREATE TABLE CRIME(
ID CRIME INT,
TIPO VARCHAR(30),
LOCAL VARCHAR(30),
DATA DATE,
);

CREATE TABLE ARMA(
ID ARMA INT,
CALIBRE INT,
MODELO VARCHAR(20),
FABRICANTE VARCHAR(10).
);
